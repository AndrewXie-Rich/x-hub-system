import turtle
import random

WIDTH = 600
HEIGHT = 700
LANES = [-170, -90, -10, 70, 150]
BASE_SPEED = 6
MAX_SPEED = 14
SPEED_BOOST_EVERY = 300
SPAWN_BASE_DELAY = 28

screen = turtle.Screen()
screen.title('Racing Game - Turtle')
screen.setup(WIDTH, HEIGHT)
screen.bgcolor('#101820')
screen.tracer(0)

hud = turtle.Turtle(visible=False)
hud.color('white')
hud.penup()
hud.goto(0, HEIGHT // 2 - 35)

road = turtle.Turtle(visible=False)
road.pensize(3)
road.speed(0)

car = turtle.Turtle()
car.shape('square')
car.shapesize(stretch_wid=1.2, stretch_len=2.0)
car.color('deepskyblue')
car.penup()
car.setposition(LANES[2], -HEIGHT // 2 + 110)

game_over_text = turtle.Turtle(visible=False)
game_over_text.color('yellow')
game_over_text.penup()
game_over_text.goto(0, 20)

obstacles = []
score = 0
speed = BASE_SPEED
frame_count = 0
game_over = False


def draw_track():
    road.clear()
    left_edge = LANES[0] - 55
    right_edge = LANES[-1] + 55

    road.color('#444')
    road.penup()
    road.goto(left_edge, -HEIGHT // 2 + 60)
    road.pendown()
    road.goto(left_edge, HEIGHT // 2 - 60)
    road.penup()
    road.goto(right_edge, -HEIGHT // 2 + 60)
    road.pendown()
    road.goto(right_edge, HEIGHT // 2 - 60)
    road.penup()

    road.color('#666')
    road.pensize(2)
    for x in LANES:
        road.penup()
        road.setheading(270)
        road.goto(x, HEIGHT // 2 - 60)
        road.pendown()
        for _ in range(22):
            road.forward(10)
            road.penup()
            road.forward(10)
            road.pendown()
    road.penup()


def lane_index(x):
    return min(range(len(LANES)), key=lambda i: abs(LANES[i] - x))


def move_left():
    global game_over
    if game_over:
        return
    idx = lane_index(car.xcor())
    if idx > 0:
        car.setx(LANES[idx - 1])


def move_right():
    global game_over
    if game_over:
        return
    idx = lane_index(car.xcor())
    if idx < len(LANES) - 1:
        car.setx(LANES[idx + 1])


def spawn_obstacle():
    obs = turtle.Turtle()
    obs.shape('square')
    obs.shapesize(stretch_wid=1.2, stretch_len=1.9)
    obs.color(random.choice(['red', 'orange', 'magenta', 'gold']))
    obs.penup()
    obs.setposition(random.choice(LANES), HEIGHT // 2 + random.randint(20, 140))
    obstacles.append(obs)


def update_hud():
    hud.clear()
    if game_over:
        hud.write('Game Over - press R to restart', align='center', font=('Arial', 14, 'bold'))
    else:
        hud.write(f'Score: {score} | Speed: {speed:.1f}', align='center', font=('Arial', 14, 'bold'))


def game_over_screen():
    global game_over
    game_over = True
    game_over_text.clear()
    game_over_text.write('GAME OVER', align='center', font=('Arial', 32, 'bold'))
    update_hud()


def step():
    global frame_count, score, speed, game_over
    if game_over:
        return

    frame_count += 1
    score += 1

    if score % SPEED_BOOST_EVERY == 0:
        speed = min(MAX_SPEED, speed + 0.5)

    spawn_delay = max(6, int(SPAWN_BASE_DELAY - (speed - BASE_SPEED) * 1.6))
    if frame_count % spawn_delay == 0:
        spawn_obstacle()

    for obs in list(obstacles):
        obs.sety(obs.ycor() - speed)
        if obs.ycor() < -HEIGHT // 2 - 40:
            obs.hideturtle()
            obstacles.remove(obs)
        elif abs(obs.xcor() - car.xcor()) < 45 and abs(obs.ycor() - car.ycor()) < 28:
            game_over_screen()
            return

    update_hud()
    screen.update()
    screen.ontimer(step, 16)


def restart():
    global score, speed, frame_count, game_over
    game_over = False
    score = 0
    speed = BASE_SPEED
    frame_count = 0
    car.setx(LANES[2])

    for obs in list(obstacles):
        obs.hideturtle()
    obstacles.clear()

    game_over_text.clear()
    draw_track()
    update_hud()
    step()


screen.listen()
screen.onkey(move_left, 'Left')
screen.onkey(move_right, 'Right')
screen.onkey(restart, 'r')
screen.onkey(restart, 'R')

draw_track()
update_hud()
step()
screen.mainloop()
